深度优先 DFS
```js
const dfs = (root) => {
    console.log(root.val);
    root.children.forEach(dfs)
}
```


广度优先 BFS
```js
const bfs = (root) => {
    const q = [root];
    while(q.length) {
        const head = q.shift();
        console.log(head.val);
        head.children.forEach(child => q.push(child))
    }
}
```


二叉树
先序遍历
```js
const pre = (root) => {
    if(!root) return;
    console.log(root.val);
    pre(root.left); 
    pre(root.right); 
}
// 或者
const pre = (root) => {
    if(!root) return;
    const stack = [root];
    while(stack.length) {
        const head = stack.pop();
        console.log(head.val);
        if(head.right) stack.push(head.right);
        if(head.left) stack.push(head.left);
    }
}
```
中序遍历
```js
const in = (root) => {
    if(!root) return;
    pre(root.left); 
    console.log(root.val);
    pre(root.right); 
}
// 或者
const in = (root) => {
    if(!root) return;
    const stack = [];
    let p = root;
    while(stack.length || p) {
        while(p) {
            stack.push(p);
            p = p.left;
        }
        const n = stack.pop();
        console.log(n.val);
        p = p.right;
    }
}
```
后序遍历
```js
const post = (root) => {
    if(!root) return;
    pre(root.left);
    pre(root.right); 
    console.log(root.val);
}
// or
const post = (root) => {
    if(!root) return;
    const stack = [root];
    const outStack = [];
    while(stack.length) {
        const n = stack.pop();
        outStack.push(n);
        if(n.left) stack.push(n.left);
        if(n.right) stack.push(n.right);
    }
    while(outStack.length) {
        const n = outStack.pop();
        console.log(n.val);
    }
}
```
